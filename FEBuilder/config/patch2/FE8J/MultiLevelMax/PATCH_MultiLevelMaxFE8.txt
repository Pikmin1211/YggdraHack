//STRUCTパッチとして実装します

//データへのポインタ

//データサイズ(10進数)

//データ個数 00 が出てくるまで
DATACOUNT=$GREP4 0x00 0x00 0x00 0x00

//リストに表示されるのはユニット名です
LISTNAME=$B0:UNIT

//データを定義します
B0:UNIT=Unit(0xFF=ANY)
B1:CLASS=Class(0xFF=ANY)
B2:DECIMAL=MaxLevel
B3=00
